@pytest.mark.asyncio
async def test_pipeline_dedup_sets_duplicate_of(db_engine):
    """Second ingest of identical text marks duplicate_of on the second source."""
    from app.core.database import AsyncSessionLocal
    from app.models.models import Source
    import hashlib

    text = "Unique text for dedup test — Phase 2 integration"
    content_hash = hashlib.sha256(text.encode()).hexdigest()

    # Create first source manually (simulating a completed ingest)
    async with AsyncSessionLocal() as session:
        s1 = Source(course_id=1, source_type="text", raw_text=text,
                    content_hash=content_hash, status="done")
        session.add(s1)
        await session.commit()
        s1_id = s1.id

    # Create second source with identical content
    async with AsyncSessionLocal() as session:
        s2 = Source(course_id=1, source_type="text", raw_text=text, status="pending")
        session.add(s2)
        await session.commit()
        s2_id = s2.id

    # Run pipeline on the second source — should detect duplicate
    from app.pipeline.pipeline import run_pipeline
    await run_pipeline(source_id=s2_id)

    # Verify second source is done (not error) with duplicate_of set
    async with AsyncSessionLocal() as session:
        result = await session.execute(sa.select(Source).where(Source.id == s2_id))
        s2 = result.scalar_one()
        assert s2.status == "done", f"Expected done, got {s2.status}"
        assert s2.source_metadata is not None
        assert "duplicate_of" in s2.source_metadata
        assert s2.source_metadata["duplicate_of"] == s1_id


@pytest.mark.asyncio
async def test_pipeline_text_creates_chunks(db_engine):
    """Pipeline creates Chunk rows for a text source."""
    from app.core.database import AsyncSessionLocal
    from app.models.models import Source, Chunk

    text = "Backpropagation computes gradients via the chain rule. " * 5  # > 50 chars
    async with AsyncSessionLocal() as session:
        source = Source(course_id=1, source_type="text", raw_text=text, status="pending")
        session.add(source)
        await session.commit()
        source_id = source.id

    from app.pipeline.pipeline import run_pipeline
    await run_pipeline(source_id=source_id)

    async with AsyncSessionLocal() as session:
        result = await session.execute(
            sa.select(sa.func.count()).select_from(Chunk).where(Chunk.source_id == source_id)
        )
        chunk_count = result.scalar()
        assert chunk_count >= 1, f"Expected at least 1 chunk, got {chunk_count}"

        # Verify source is done
        src_result = await session.execute(sa.select(Source).where(Source.id == source_id))
        source = src_result.scalar_one()
        assert source.status == "done"
