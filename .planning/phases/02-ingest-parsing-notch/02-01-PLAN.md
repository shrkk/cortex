```python
"""add courses.embedding vector column

Revision ID: 0002
Revises: 0001
Create Date: 2026-04-25

"""
from alembic import op
import sqlalchemy as sa
from pgvector.sqlalchemy import Vector

revision = "0002"
down_revision = "0001"
branch_labels = None
depends_on = None


def upgrade() -> None:
    op.add_column(
        "courses",
        sa.Column("embedding", Vector(1536), nullable=True),
    )


def downgrade() -> None:
    op.drop_column("courses", "embedding")
```

Critical notes:
- `down_revision = "0001"` — links to the initial migration
- Column is nullable=True — existing rows have no embedding until the seed backfill runs (Plan 02-05)
- No index on courses.embedding — the table is tiny (<100 rows); sequential scan is fine and hnsw on a near-empty table has no benefit
- Do NOT call `op.execute("CREATE EXTENSION IF NOT EXISTS vector")` — already done in 0001
