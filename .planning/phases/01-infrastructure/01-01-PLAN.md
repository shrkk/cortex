`backend/tests/test_migration.py`:
```python
import pytest
from sqlalchemy import inspect, text
from sqlalchemy.ext.asyncio import create_async_engine
from app.core.config import settings

EXPECTED_TABLES = {
    "users",
    "courses",
    "sources",
    "chunks",
    "concepts",
    "concept_sources",
    "extraction_cache",
    "edges",
    "flashcards",
    "quizzes",
}


async def test_all_tables_exist():
    engine = create_async_engine(settings.database_url)
    async with engine.connect() as conn:
        table_names = await conn.run_sync(
            lambda sync_conn: inspect(sync_conn).get_table_names()
        )
    await engine.dispose()
    missing = EXPECTED_TABLES - set(table_names)
    assert missing == set(), f"Missing tables: {missing}"


async def test_vector_extension_installed():
    engine = create_async_engine(settings.database_url)
    async with engine.connect() as conn:
        result = await conn.execute(
            text("SELECT extname FROM pg_extension WHERE extname = 'vector'")
        )
        row = result.fetchone()
    await engine.dispose()
    assert row is not None, "pgvector extension not installed"


async def test_hnsw_index_exists():
    engine = create_async_engine(settings.database_url)
    async with engine.connect() as conn:
        result = await conn.execute(
            text(
                "SELECT indexname FROM pg_indexes "
                "WHERE tablename = 'concepts' AND indexname = 'concepts_embedding_idx'"
            )
        )
        row = result.fetchone()
    await engine.dispose()
    assert row is not None, "hnsw index concepts_embedding_idx not found on concepts table"
```

`backend/tests/test_seed.py`:
```python
import pytest
from sqlalchemy import text
from sqlalchemy.ext.asyncio import create_async_engine
from app.core.config import settings


async def test_seed_creates_user_and_courses():
    engine = create_async_engine(settings.database_url)
    async with engine.connect() as conn:
        user_result = await conn.execute(
            text("SELECT id FROM users WHERE id = 1")
        )
        user_row = user_result.fetchone()

        course_result = await conn.execute(
            text("SELECT COUNT(*) FROM courses WHERE user_id = 1")
        )
        course_count = course_result.scalar()
    await engine.dispose()

    assert user_row is not None, "user_id=1 not found in users table"
    assert course_count >= 1, f"Expected at least 1 course for user_id=1, got {course_count}"
```
